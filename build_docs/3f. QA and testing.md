## 1. Testing Philosophy

NyayaSetu is a legal intelligence product. A wrong output does not just frustrate a user — it may cause a citizen to take incorrect legal action, miss a critical clause, or walk into a police station with a flawed document. The cost of a false output in this system is categorically higher than in a typical SaaS product.

This testing plan is built around three non-negotiable guarantees:

**Guarantee 1 — Zero hallucination.** The system must never reference a clause that was not retrieved from the corpus. This is tested at the node level, the pipeline level, and in production sampling.

**Guarantee 2 — Correct escalation.** The system must never return a low-confidence or high-stakes result directly to a citizen without routing through human review. Escalation logic is tested exhaustively across boundary conditions.

**Guarantee 3 — Payment integrity.** A writeup must never be generated without confirmed payment. Payment and writeup flows are tested for every failure mode including network interruption, signature mismatch, and race conditions.

Every other test category supports these three guarantees.

# tests/backend/unit/test_nodes/test_confidence_evaluator.py

class TestConfidenceEvaluator:

    def make_state_with_clauses(self, scores, high_stakes_raw=None, raw_input=""):
        clauses = [
            {"act": "BNS", "section_number": str(i),
             "relevance_score": s, "source": "qdrant"}
            for i, s in enumerate(scores)
        ]
        return {
            "merged_clauses": clauses,
            "_high_stakes_raw": high_stakes_raw or [],
            "raw_input": raw_input,
            "errors": []
        }

    def test_high_confidence_proceeds(self):
        state = self.make_state_with_clauses([0.92, 0.89, 0.85])
        result = confidence_evaluator_node(state)
        assert result["routing_decision"] == "proceed"
        assert result["confidence_score"] >= 0.85

    def test_low_confidence_escalates(self):
        state = self.make_state_with_clauses([0.35, 0.30, 0.28])
        result = confidence_evaluator_node(state)
        assert result["routing_decision"] == "escalate"

    def test_high_stakes_flag_forces_escalation(self):
        # Even with high confidence, high stakes must escalate
        state = self.make_state_with_clauses(
            [0.95, 0.92, 0.90],
            high_stakes_raw=["life_at_risk"]
        )
        result = confidence_evaluator_node(state)
        assert result["routing_decision"] == "escalate"
        assert result["high_stakes_flag"] == "life_at_risk"

    def test_high_stakes_keyword_in_hindi_escalates(self):
        state = self.make_state_with_clauses(
            [0.90, 0.88, 0.85],
            raw_input="meri beti ko kidnap kar liya"  # kidnap keyword
        )
        result = confidence_evaluator_node(state)
        assert result["routing_decision"] == "escalate"

    def test_empty_results_produces_zero_confidence(self):
        state = self.make_state_with_clauses([])
        result = confidence_evaluator_node(state)
        assert result["confidence_score"] == 0.0
        assert result["routing_decision"] == "escalate"

    def test_confidence_score_boundary_at_threshold(self):
        # Exactly at threshold — should escalate (not proceed)
        state = self.make_state_with_clauses([0.45, 0.45, 0.45])
        result = confidence_evaluator_node(state)
        # 0.45 is the escalation threshold — at or below escalates
        assert result["routing_decision"] == "escalate"

    def test_one_above_threshold_proceeds(self):
        state = self.make_state_with_clauses([0.46, 0.46, 0.46])
        result = confidence_evaluator_node(state)
        assert result["routing_decision"] == "proceed"

        result = await response_synthesiser_node(state)
        # Hallucinated clause must be blocked
        assert len(result["synthesised_clauses"]) == 1
        assert result["synthesised_clauses"][0]["section_number"] == "281"

import time

class TestPerformanceBenchmarks:

    @pytest.mark.asyncio
    @pytest.mark.performance
    async def test_query_response_p95_under_15_seconds(self):
        """
        Run 20 queries concurrently, measure p95 response time.
        Must be under 15 seconds.
        Requires: live pipeline with Qdrant + Neo4j + Groq
        """
        from pipeline.graph import run_pipeline

        queries = [
            "Someone stole my phone",
            "I was assaulted on the street",
            "Police refused to file FIR",
            "My landlord is harassing me",
            "Someone forged my documents",
        ] * 4  # 20 queries total

        async def timed_query(i, q):
            start = time.monotonic()
            await run_pipeline(query_id=f"perf-{i:03d}", raw_input=q)
            return time.monotonic() - start

        tasks = [timed_query(i, q) for i, q in enumerate(queries)]
        durations = await asyncio.gather(*tasks)

        durations_sorted = sorted(durations)
        p95 = durations_sorted[int(0.95 * len(durations_sorted))]
        p50 = statistics.median(durations)

        print(f"\nQuery Performance: p50={p50:.2f}s p95={p95:.2f}s")
